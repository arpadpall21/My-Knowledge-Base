Data Types 

    Type Annotation             Example                     Usage
    
    any                           // any data type
    null                        null                                         // null value
    
    
    string                      "Hello World!"              string    // string (between double quotes)
    number                      123.123                     number    // positive or negative float
    bool                        true|false                  bool      // boolean (true|false)
    
    
    list(type)                  ["one", "two"]              list[idx]    // ordered collection of the same type
    set(type)                   ["one", "two"]              [no indexing]    // unordered collection of the same type holding unique values    
                                                      // -! members are NOT accessible individually (we have to use the for_each meta argument)
    tuple([type, ...])          ["one", 2]                  tuple[idx]               // ordered collection of different types
    
    
    map(type)                   {                           map["key"] | map.key             // key/value pairs of the same type
                                  key1 = "one"
                                  key2 = "two"
                                }
    object({                    {                           object["key"] | object.key         // key/value pairs of different types
      key = type,                 key1 = "one"
      ...                         key2 = 2
    })                          }
DEMO (can be pseudocode)

string and number data types 

// main.tf (file) ---------------------------------------
    resource "local_file" "testFile" {
      filename = var.filePath
      content  = var.fileContent
    }
  
  
// variables.tf (file) ----------------------------------
    variable "filePath" {
      type    = string
      default = "./myFile.txt"
    }
  
    variable "fileContent" {
      type    = number
      default = 123.0123
    }

list data type 

// main.tf (file) ---------------------------------------
    resource "local_file" "testFile" {
      filename = var.file[0]
      content  = var.file[1]
    }
  
  
// variables.tf (file) ----------------------------------
    variable "file" {
      type    = list(string)
      default = ["./myFile.txt", "Hello World!"]
    }

set data type 

// main.tf (file) ---------------------------------------
    resource "local_file" "testFile2" {
      for_each = var.someSet
      
      filename = each.key                 // members are accessible through iteration 
      content = "File content"
    }
    
    
// variables.tf (file) ----------------------------------
    variable "someSet" {
      type    = set(string)
      default = ["file_1.txt", "file_2.txt"]
    }

tuple data type 

// main.tf (file) ---------------------------------------
    resource "local_file" "testFile" {
      filename = var.file[0]
      content  = var.file[1]
    }
  
  
// variables.tf (file) ----------------------------------
    variable "file" {
      type    = tuple([string, number])
      default = ["./myFile.txt", 123.0233]
    }

map data type 

// main.tf (file) ---------------------------------------
    resource "local_file" "testFile" {
      filename = var.file.path
      content  = var.file.content
    }
  
  
// variables.tf (file) ----------------------------------
    variable "file" {
      type = map(string)
      default = {
        path    = "./someFile.txt"
        content = "Hello World!"
      }
    }

object data type 

// main.tf (file) ---------------------------------------
    resource "local_file" "testFile" {
      filename = var.file.path
      content  = var.file.content
    }
  
  
// variables.tf (file) ----------------------------------
    variable "file" {
      type = object({
        path   = string
        content = number
      })
      default = {
        path    = "./someFile.txt"
        content = 123.321
      }
    }
